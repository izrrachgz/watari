using System.Collections.Generic;
using System.Threading.Tasks;
using Contexto.Entidades;
using Contexto.ProveedoresDeDatos;
using Utilidades.Modelos;

namespace Negocio.ProveedoresDeDatos
{
  /// <summary>
  /// Provee los mecanismos para acceder a los datos
  /// de la entidad {{Nombre}}
  /// </summary>
  public class Proveedor{{Nombre}} : ProveedorDeDatosBase<{{Nombre}}>
  {
    /// <summary>
    /// Devuelve todos los registros de {{Nombre}} dentro de un modelo de datos paginado
    /// </summary>
    /// <param name="pagina">Solicitud de Pagina</param>
    /// <returns>Coleccion de elementos de {{Nombre}}</returns>
    public async Task<RespuestaColeccion<{{Nombre}}>> Paginado(Paginado pagina)
      => await base.Obtener(pagina);

    /// <summary>
    /// Devuelve un registro de {{Nombre}} cuyo identificador primario
    /// es igual al proporcionado
    /// </summary>
    /// <param name="id">Identificador Primario de {{Nombre}}</param>
    /// <returns>Elemento de {{Nombre}}</returns>
    public async Task<RespuestaModelo<{{Nombre}}>> ObtenerPorId(int id)
      => await base.Obtener(id);
    
    /// <summary>
    /// Guarda los cambios efectuados de la entidad
    /// nuevos/actualizacion
    /// </summary>
    /// <param name="modelo">Entidad</param>
    /// <returns>Respuesta basica</returns>
    public async Task<RespuestaBasica> Guardar{{Nombre}}({{Nombre}} modelo)
      => await base.Guardar(modelo);
  }
}
